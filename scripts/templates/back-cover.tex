% Plain Dharma — XeLaTeX back cover, 6×9" (2:3), to pair with the front cover.
%
% The front cover is a designer's InDesign PDF (scripts/assets/PlainDharma_Cover.pdf)
% that generate-cover.ts merely rasterizes. There is no InDesign back cover, so
% this file IS the back-cover source: a self-contained XeLaTeX page that
% generate-back-cover.ts compiles → PDF, then runs through the same
% pdftoppm → magick rasterizer the front cover uses.
%
% Palette + fonts match scripts/templates/pdf-preamble.tex and the site tokens.
% The gold "stitched" stripe mirrors the front cover's binding edge — on the
% FRONT it sits at the left (spine) edge; on the BACK the spine is the right
% edge, so the stripe lives on the right and the two covers mirror across it.
%
% This file is a TEMPLATE: generate-back-cover.ts substitutes the __TOKENS__
% per target (screen 6×9, or print 5.25×8.25 = 5×8 trim + 0.125in bleed) so one
% layout drives every trim. __FONTS_DIR__ becomes the absolute fonts dir
% (fontspec's Path= won't reliably expand a relative path); the geometry/stripe/
% overlay tokens shrink the page and inset the artwork to clear the print bleed.
\documentclass[__FONTSIZE__]{article}

\usepackage{fontspec}
\usepackage[paperwidth=__PAPER_W__,paperheight=__PAPER_H__,top=__M_TOP__,bottom=__M_BOT__,left=__M_LEFT__,right=__M_RIGHT__]{geometry}
\usepackage{xcolor}
\usepackage{tikz}
\usepackage{eso-pic}
\usepackage{newunicodechar}

% Brand palette — matches the site's --color-* tokens and pdf-preamble.tex.
% Stripe gold + stitch saffron are sampled from the rasterized front cover so
% the pair matches exactly (#FBC608 band, #F26C15 dashed stitch).
\definecolor{cream}{HTML}{F5EFE0}
\definecolor{ink}{HTML}{2A2520}
\definecolor{muted}{HTML}{6B5F50}
\definecolor{saffron}{HTML}{F26C15}
\definecolor{gold}{HTML}{FBC608}
\definecolor{boxborder}{HTML}{D8CBAE}

% Gold stitched stripe on the RIGHT (spine) edge + dashed saffron stitch.
% Band mirrors the front cover; stitch sits 0.07in inside the band's inner edge.
% On print trims the band is widened so it still reads ~0.5in after the bleed is
% guillotined off the right. current-page node needs two passes.
\newlength{\stripeW}\setlength{\stripeW}{__STRIPE_W__}
\newlength{\stitchX}\setlength{\stitchX}{__STITCH_X__}
\AddToShipoutPictureBG{%
  \begin{tikzpicture}[remember picture,overlay]
    \fill[gold]
      ([xshift=-\stripeW]current page.north east) rectangle (current page.south east);
    \draw[saffron,line width=1.4pt,dash pattern=on 9pt off 6pt]
      ([xshift=-\stitchX]current page.north east) -- ([xshift=-\stitchX]current page.south east);
  \end{tikzpicture}%
}

% Bottom block (site + license + formats, and the ISBN/barcode placeholder),
% drawn as a page overlay anchored to the bottom edge rather than placed in the
% text flow. In-flow placement fought LaTeX's page-break algorithm (the \vspace
% before it is a legal breakpoint and the block spilled to a 2nd page even with
% room to spare); anchoring guarantees one fixed-layout page. Insets match the
% text margins (0.9in left); the barcode sits clear of the right-edge stripe.
% The flow content above is tuned to end above this block so they never collide.
\AddToShipoutPictureFG{%
  \begin{tikzpicture}[remember picture,overlay]
    \node[anchor=south west,inner sep=0pt] at
      ([xshift=__OV_LEFT__,yshift=__OV_Y__]current page.south west) {%
      \begin{minipage}[b]{__OV_TEXTW__}

      \end{minipage}%
    };
    \node[anchor=south east,inner sep=0pt] at
      ([xshift=-__OV_RIGHT__,yshift=__OV_Y__]current page.south east) {%
      \fcolorbox{boxborder}{white}{%
        \begin{minipage}[c][0.62in][c]{1.05in}

  \end{tikzpicture}%
}
